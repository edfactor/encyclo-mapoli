--this compares counts in SMART schema TEAM_PROFIT_SHARING2
--                        Ready schema PROFITSHARE

SELECT 
    (SELECT COUNT(*) FROM TEAM_PROFIT_SHARING2.DEMOGRAPHIC) AS SMART_DEMOGRAPHIC,
    (SELECT COUNT(*) FROM PROFITSHARE.DEMOGRAPHICS) AS READY_DEMOGRAPHICS,
    (SELECT COUNT(*) FROM TEAM_PROFIT_SHARING2.BENEFICIARY) AS SMART_BENEFICIARY,
    (SELECT COUNT(*) FROM PROFITSHARE.PAYBEN) AS READY_PAYBEN,
    (SELECT COUNT(*) FROM TEAM_PROFIT_SHARING2.PROFIT_DETAIL) AS SMART_PROFIT_DETAIL,
    (SELECT COUNT(*) FROM PROFITSHARE.PROFIT_DETAIL) + 
    (SELECT COUNT(*) FROM PROFITSHARE.PROFIT_SS_DETAIL) AS READY_PROFIT_DETAIL_SUM,
    (SELECT COUNT(*) FROM TEAM_PROFIT_SHARING2.PAY_PROFIT) AS SMART_PAY_PROFIT,
    (SELECT COUNT(*) FROM PROFITSHARE.PAYPROFIT) AS READY_PAYPROFIT,
    (SELECT COUNT(*) FROM TEAM_PROFIT_SHARING2.DISTRIBUTION) AS SMART_DISTRIBUTION,
    (SELECT COUNT(*) FROM PROFITSHARE.PROFDIST) AS READY_PROFDIST
FROM dual;
